```bash
# 基本语法：
# gh workflow run <workflow-file> -f <参数名>=<值> [其他参数]

# ============================================
# 场景1：仅构建Docker镜像（不发布）
# ============================================
gh workflow run next-release.yml \
  -f release=false \
  -f platforms="none" \
  -f build_docker=true \
  -f docker_registries="ghcr.io,docker.io" \
  -f use_china_mirror=false \
  -f optimize_binary=true \
  -f optimization_level="9"

# 简写版本：
gh workflow run next-release.yml -f release=false -f platforms="none" -f build_docker=true

# ============================================
# 场景2：仅构建Windows二进制（测试构建）
# ============================================
gh workflow run next-release.yml \
  -f release=false \
  -f platforms="windows" \
  -f build_docker=false \
  -f optimize_binary=true \
  -f optimization_level="5"

# 简写版本：
gh workflow run next-release.yml -f release=false -f platforms="windows" -f build_docker=false

# ============================================
# 场景3：仅构建macOS二进制
# ============================================
gh workflow run next-release.yml \
  -f release=false \
  -f platforms="macos" \
  -f build_docker=false \
  -f optimize_binary=false  # 测试无优化的版本

# ============================================
# 场景4：仅构建Linux二进制
# ============================================
gh workflow run next-release.yml -f release=false -f platforms="linux" -f build_docker=false -f optimize_binary=true -f optimization_level="9"

# ============================================
# 场景5：构建所有平台但不发布（完整测试）
# ============================================
gh workflow run next-release.yml \
  -f release=false \
  -f platforms="linux,windows,macos" \
  -f build_docker=true \
  -f docker_registries="ghcr.io" \
  -f optimize_binary=true

# ============================================
# 场景6：创建正式发布（tag触发时使用）
# ============================================
# 注意：通常通过git tag自动触发，但也可以手动模拟
gh workflow run next-release.yml \
  -f release=true \
  -f platforms="linux,windows,macos" \
  -f build_docker=true \
  -f docker_registries="ghcr.io,docker.io" \
  -f optimize_binary=true \
  -f optimization_level="9"

# ============================================
# 场景7：使用中国镜像加速构建
# ============================================
gh workflow run next-release.yml \
  -f release=false \
  -f platforms="linux" \
  -f build_docker=true \
  -f use_china_mirror=true \
  -f optimize_binary=true

# ============================================
# 场景8：构建特定架构的macOS
# ============================================
gh workflow run next-release.yml \
  -f release=false \
  -f platforms="macos" \
  -f build_docker=false \
  -f optimize_binary=true \
  -f optimization_level="7"

# 你可以通过环境变量指定架构偏好：
# 在仓库设置中添加变量：PLATFORMS="macos"（只构建当前系统架构）
# 或使用：PLATFORMS="macos-x86_64,macos-arm64"

# ============================================
# 场景9：仅构建Docker镜像到特定注册表
# ============================================
# 只推送到GHCR
gh workflow run next-release.yml \
  -f release=false \
  -f platforms="none" \
  -f build_docker=true \
  -f docker_registries="ghcr.io"

# 只推送到Docker Hub
gh workflow run next-release.yml \
  -f release=false \
  -f platforms="none" \
  -f build_docker=true \
  -f docker_registries="docker.io"

# ============================================
# 场景10：测试低优化级别
# ============================================
gh workflow run next-release.yml \
  -f release=false \
  -f platforms="linux,windows" \
  -f build_docker=false \
  -f optimize_binary=true \
  -f optimization_level="1"  # 最快压缩，文件稍大
```
